## 2. Metric 三层基类设计

### 2.1 基类定义

**文件**: `deepeval/metrics/base_metric.py`

```python
from abc import ABC, abstractmethod
from typing import Optional
from pydantic import BaseModel

class BaseMetric(ABC):
    """所有指标的抽象基类"""

    # 评测配置
    threshold: float                      # 通过阈值
    async_mode: bool = True               # 是否异步
    verbose_mode: bool = True             # 详细日志
    include_reason: bool = False          # 包含解释

    # 评测结果
    score: Optional[float] = None
    reason: Optional[str] = None
    success: Optional[bool] = None
    evaluation_model: Optional[str] = None
    evaluation_cost: Optional[float] = None
    verbose_logs: Optional[str] = None

    @abstractmethod
    def measure(self, test_case: "LLMTestCase") -> float:
        """同步评测"""
        pass

    @abstractmethod
    async def a_measure(self, test_case: "LLMTestCase") -> float:
        """异步评测"""
        pass

    @abstractmethod
    def is_successful(self) -> bool:
        """判断是否通过"""
        pass

    def _accrue_cost(self, cost: float):
        """累计 API 调用成本"""
        if self.evaluation_cost is None:
            self.evaluation_cost = 0
        self.evaluation_cost += cost


class BaseConversationalMetric(ABC):
    """多轮对话评测基类"""

    @abstractmethod
    def measure(self, test_case: "ConversationalTestCase") -> float:
        pass


class BaseArenaMetric(ABC):
    """A/B 对比评测基类 - 返回胜出方而非分数"""

    @abstractmethod
    def measure(self, test_case: "ArenaTestCase") -> str:
        """返回 'output_a', 'output_b', 或 'tie'"""
        pass
```

### 2.2 三层设计的优势

| 基类 | 适用场景 | 返回值 |
|-----|---------|--------|
| `BaseMetric` | 单轮问答评测 | `float` (0-1) |
| `BaseConversationalMetric` | 多轮对话评测 | `float` (0-1) |
| `BaseArenaMetric` | 两个输出对比 | `str` (winner) |

### 3.2 Template - Prompt 模板

**文件**: `deepeval/metrics/faithfulness/template.py`

```python
import textwrap

class FaithfulnessTemplate:
    """忠实度评测的 Prompt 模板"""

    @staticmethod
    def generate_claims(actual_output: str) -> str:
        return textwrap.dedent(f"""
            Based on the given text, identify and list all the claims made.
            A claim is a statement that asserts something to be true.

            Text:
            {actual_output}

            Output a JSON object with a single key "claims" containing a list of strings.
        """)

    @staticmethod
    def generate_truths(retrieval_context: List[str]) -> str:
        context_str = "\n---\n".join(retrieval_context)
        return textwrap.dedent(f"""
            Based on the given context, extract all factual statements.

            Context:
            {context_str}

            Output a JSON object with a single key "truths" containing a list of strings.
        """)

    @staticmethod
    def generate_verdicts(claims: List[str], truths: List[str]) -> str:
        claims_str = "\n".join(f"- {c}" for c in claims)
        truths_str = "\n".join(f"- {t}" for t in truths)
        return textwrap.dedent(f"""
            For each claim, determine if it is supported by the truths.

            Claims:
            {claims_str}

            Truths:
            {truths_str}

            For each claim, output:
            - "yes" if the claim is fully supported
            - "no" if the claim contradicts the truths
            - "idk" if there's not enough information

            Output a JSON object with key "verdicts" containing a list of objects,
            each with "verdict" and optional "reason" fields.
        """)
```

### 3.3 Implementation - 指标实现

**文件**: `deepeval/metrics/faithfulness/faithfulness.py`

```python
from deepeval.metrics.base_metric import BaseMetric
from deepeval.metrics.faithfulness.schema import Claims, Truths, Verdicts
from deepeval.metrics.faithfulness.template import FaithfulnessTemplate
from deepeval.metrics.utils import generate_with_schema_and_extract

class FaithfulnessMetric(BaseMetric):
    """忠实度评测 - 检查输出是否基于给定上下文"""

    _required_params = [
        LLMTestCaseParams.INPUT,
        LLMTestCaseParams.ACTUAL_OUTPUT,
        LLMTestCaseParams.RETRIEVAL_CONTEXT,
    ]

    def __init__(
        self,
        threshold: float = 0.5,
        model: Optional[str] = None,
        include_reason: bool = False,
        async_mode: bool = True,
        truths_extraction_limit: Optional[int] = None,
    ):
        self.threshold = threshold
        self.model = model or get_default_model()
        self.include_reason = include_reason
        self.async_mode = async_mode
        self.truths_extraction_limit = truths_extraction_limit

    async def a_measure(self, test_case: LLMTestCase) -> float:
        # Step 1: 提取事实
        truths_prompt = FaithfulnessTemplate.generate_truths(
            test_case.retrieval_context
        )
        truths: Truths = await generate_with_schema_and_extract(
            self.model, truths_prompt, Truths
        )

        # Step 2: 提取陈述
        claims_prompt = FaithfulnessTemplate.generate_claims(
            test_case.actual_output
        )
        claims: Claims = await generate_with_schema_and_extract(
            self.model, claims_prompt, Claims
        )

        # Step 3: 判定每条陈述
        verdicts_prompt = FaithfulnessTemplate.generate_verdicts(
            claims.claims, truths.truths
        )
        verdicts: Verdicts = await generate_with_schema_and_extract(
            self.model, verdicts_prompt, Verdicts
        )

        # Step 4: 计算得分
        yes_count = sum(1 for v in verdicts.verdicts if v.verdict == "yes")
        self.score = yes_count / len(verdicts.verdicts) if verdicts.verdicts else 0
        self.success = self.score >= self.threshold

        return self.score

    def measure(self, test_case: LLMTestCase) -> float:
        """同步版本 - 包装异步实现"""
        loop = get_or_create_event_loop()
        return loop.run_until_complete(self.a_measure(test_case))

    def is_successful(self) -> bool:
        return self.success
```

## 4. LLM Provider 抽象

### 4.1 基类设计

**文件**: `deepeval/models/base_model.py`

```python
from abc import ABC, abstractmethod
from typing import Optional, Type, TypeVar

T = TypeVar("T", bound=BaseModel)

class DeepEvalBaseLLM(ABC):
    """LLM Provider 抽象基类"""

    def __init__(self, model: Optional[str] = None):
        self.model_name = model

    @abstractmethod
    def generate(self, prompt: str, **kwargs) -> str:
        """同步生成"""
        pass

    @abstractmethod
    async def a_generate(self, prompt: str, **kwargs) -> str:
        """异步生成"""
        pass

    def generate_with_schema(
        self,
        prompt: str,
        schema: Type[T],
        **kwargs
    ) -> T:
        """结构化输出 - 尝试使用 JSON mode，失败则回退"""
        try:
            # 尝试使用原生 JSON mode
            response = self._generate_json_mode(prompt, schema, **kwargs)
        except NotImplementedError:
            # 回退到普通生成 + 解析
            response = self.generate(prompt, **kwargs)

        return self._parse_response(response, schema)

    def _parse_response(self, response: str, schema: Type[T]) -> T:
        """健壮的 JSON 解析"""
        json_data = trim_and_load_json(response)
        return schema.model_validate(json_data)
```

### 4.2 Provider 实现示例

**文件**: `deepeval/models/llms/openai_model.py`

```python
from openai import OpenAI, AsyncOpenAI

class GPTModel(DeepEvalBaseLLM):
    """OpenAI GPT 模型实现"""

    def __init__(
        self,
        model: str = "gpt-4o",
        api_key: Optional[str] = None,
        base_url: Optional[str] = None,
    ):
        super().__init__(model)
        self.client = OpenAI(api_key=api_key, base_url=base_url)
        self.async_client = AsyncOpenAI(api_key=api_key, base_url=base_url)

    def generate(self, prompt: str, **kwargs) -> str:
        response = self.client.chat.completions.create(
            model=self.model_name,
            messages=[{"role": "user", "content": prompt}],
            **kwargs
        )
        return response.choices[0].message.content

    async def a_generate(self, prompt: str, **kwargs) -> str:
        response = await self.async_client.chat.completions.create(
            model=self.model_name,
            messages=[{"role": "user", "content": prompt}],
            **kwargs
        )
        return response.choices[0].message.content

    def _generate_json_mode(self, prompt: str, schema: Type[T], **kwargs) -> str:
        """使用 OpenAI 的 JSON mode"""
        response = self.client.chat.completions.create(
            model=self.model_name,
            messages=[{"role": "user", "content": prompt}],
            response_format={"type": "json_object"},
            **kwargs
        )
        return response.choices[0].message.content
```

from pydantic import BaseModel
from typing import Optional, List, Dict, Any, Literal
from enum import Enum

class LLMTestCaseParams(str, Enum):
    """测试用例字段枚举 - 用于依赖声明"""
    INPUT = "input"
    ACTUAL_OUTPUT = "actual_output"
    EXPECTED_OUTPUT = "expected_output"
    CONTEXT = "context"
    RETRIEVAL_CONTEXT = "retrieval_context"
    TOOLS_CALLED = "tools_called"
    EXPECTED_TOOLS = "expected_tools"

class ToolCall(BaseModel):
    """工具调用记录"""
    name: str
    input_parameters: Dict[str, Any]
    output: Optional[str] = None

class LLMTestCase(BaseModel):
    """LLM 测试用例"""
    input: str                                      # 用户输入
    actual_output: Optional[str] = None             # 实际输出
    expected_output: Optional[str] = None           # 期望输出
    context: Optional[List[str]] = None             # 上下文 (幻觉检测)
    retrieval_context: Optional[List[str]] = None   # 检索上下文 (RAG)
    tools_called: Optional[List[ToolCall]] = None   # 调用的工具
    expected_tools: Optional[List[ToolCall]] = None # 期望调用的工具
    additional_metadata: Optional[Dict] = None      # 额外元数据

    def __hash__(self):
        """用于缓存 key 生成"""
        return hash((
            self.input,
            self.actual_output,
            self.expected_output,
            tuple(self.context or []),
            tuple(self.retrieval_context or []),
        ))
```

## 6. 异步执行模式

### 6.1 Event Loop 管理

**文件**: `deepeval/utils.py`

```python
import asyncio
from typing import TypeVar, Coroutine

T = TypeVar("T")

def get_or_create_event_loop() -> asyncio.AbstractEventLoop:
    """线程安全的 event loop 获取/创建"""
    try:
        # 尝试获取正在运行的 loop (Jupyter 等环境)
        loop = asyncio.get_running_loop()
    except RuntimeError:
        try:
            # 尝试获取当前线程的 loop
            loop = asyncio.get_event_loop()
            if loop.is_closed():
                raise RuntimeError("Event loop is closed")
        except RuntimeError:
            # 创建新的 loop
            loop = asyncio.new_event_loop()
            asyncio.set_event_loop(loop)
    return loop

def run_sync(coro: Coroutine[Any, Any, T]) -> T:
    """在同步上下文中运行协程"""
    loop = get_or_create_event_loop()
    return loop.run_until_complete(coro)
```

### 6.3 并发执行引擎

**文件**: `deepeval/evaluate/execute.py`

```python
import asyncio
from typing import List, Optional
from deepeval.metrics.base_metric import BaseMetric
from deepeval.test_case.llm_test_case import LLMTestCase

async def measure_metric_task(
    metric: BaseMetric,
    test_case: LLMTestCase,
    cached_result: Optional[CachedMetricData],
    ignore_errors: bool = False,
) -> None:
    """单个指标的异步评测任务"""
    # 检查缓存
    if cached_result and metric_config_matches(metric, cached_result):
        metric.score = cached_result.score
        metric.success = cached_result.success
        metric.reason = cached_result.reason
        return

    # 执行评测
    try:
        await asyncio.wait_for(
            metric.a_measure(test_case),
            timeout=settings.PER_TASK_TIMEOUT_SECONDS
        )
    except asyncio.TimeoutError:
        metric.error = "Evaluation timed out"
        metric.success = False
        if not ignore_errors:
            raise
    except Exception as e:
        metric.error = str(e)
        metric.success = False
        if not ignore_errors:
            raise

async def execute_test_cases(
    test_cases: List[LLMTestCase],
    metrics: List[BaseMetric],
    async_config: AsyncConfig,
    ignore_errors: bool = False,
) -> List[TestResult]:
    """批量执行测试用例"""
    semaphore = asyncio.Semaphore(async_config.max_concurrent)

    async def run_with_semaphore(metric, test_case, cached):
        async with semaphore:
            if async_config.throttle_value > 0:
                await asyncio.sleep(async_config.throttle_value)
            await measure_metric_task(metric, test_case, cached, ignore_errors)

    tasks = []
    for test_case in test_cases:
        cached = cache_manager.get_cached(test_case)
        for metric in metrics:
            tasks.append(run_with_semaphore(metric, test_case, cached))

    await asyncio.gather(*tasks, return_exceptions=ignore_errors)

    return collect_results(test_cases, metrics)
```

### 7.2 缓存 Key 生成

```python
def generate_cache_key(test_case: LLMTestCase, hyperparameters: Optional[str] = None) -> str:
    """生成测试用例的缓存 key"""
    data = {
        "input": test_case.input,
        "actual_output": test_case.actual_output,
        "expected_output": test_case.expected_output,
        "context": test_case.context,
        "retrieval_context": test_case.retrieval_context,
        "hyperparameters": hyperparameters,
    }
    serialized = json.dumps(data, sort_keys=True, default=str)
    return hashlib.sha256(serialized.encode()).hexdigest()
```

### 7.3 缓存管理器

```python
import portalocker
from pathlib import Path

class TestRunCacheManager:
    """测试运行缓存管理"""

    CACHE_DIR = Path(".deepeval")
    CACHE_FILE = CACHE_DIR / ".deepeval-cache.json"

    def __init__(self):
        self.CACHE_DIR.mkdir(exist_ok=True)
        self._cache = self._load_cache()

    def _load_cache(self) -> CachedTestRun:
        if self.CACHE_FILE.exists():
            with open(self.CACHE_FILE) as f:
                return CachedTestRun.model_validate_json(f.read())
        return CachedTestRun(test_cases_lookup_map={})

    def get_cached_test_case(
        self,
        test_case: LLMTestCase,
        hyperparameters: Optional[str] = None
    ) -> Optional[CachedTestCase]:
        key = generate_cache_key(test_case, hyperparameters)
        return self._cache.test_cases_lookup_map.get(key)

    def cache_test_case(
        self,
        test_case: LLMTestCase,
        metrics_data: List[CachedMetricData],
        hyperparameters: Optional[str] = None
    ):
        key = generate_cache_key(test_case, hyperparameters)

        # 原子写入 (使用文件锁)
        with portalocker.Lock(self.CACHE_FILE, "w", timeout=10) as f:
            self._cache.test_cases_lookup_map[key] = CachedTestCase(
                cached_metrics_data=metrics_data,
                hyperparameters=hyperparameters
            )
            f.write(self._cache.model_dump_json(indent=2))

    def metric_config_matches(
        self,
        metric: BaseMetric,
        cached_data: CachedMetricData
    ) -> bool:
        """检查指标配置是否匹配缓存"""
        return (
            metric.__class__.__name__ == cached_data.metric_name
            and metric.threshold == cached_data.metric_configuration.threshold
            and metric.model == cached_data.metric_configuration.model
        )
```

## 8. JSON 健壮解析

### 8.1 问题场景

LLM 返回的 JSON 经常有问题：
- 包含 Markdown 代码块: ` ```json ... ``` `
- 包含额外文字解释
- 转义字符问题
- 不完整的 JSON

### 8.2 解决方案

**文件**: `deepeval/metrics/utils.py`

```python
import re
import json
from typing import Any

def trim_and_load_json(response: str) -> Any:
    """健壮的 JSON 提取和解析"""

    # Step 1: 去除 Markdown 代码块
    response = re.sub(r"```json\s*", "", response)
    response = re.sub(r"```\s*$", "", response)
    response = re.sub(r"```", "", response)

    # Step 2: 尝试找到 JSON 对象/数组的边界
    json_match = re.search(r"[\[{].*[\]}]", response, re.DOTALL)
    if json_match:
        response = json_match.group()

    # Step 3: 修复常见转义问题
    response = response.replace("\n", "\\n")
    response = re.sub(r'(?<!\\)"(?=\w)', '\\"', response)  # 未转义的引号

    # Step 4: 尝试解析
    try:
        return json.loads(response)
    except json.JSONDecodeError:
        # Step 5: 更激进的清理
        response = re.sub(r"[\x00-\x1f\x7f-\x9f]", "", response)
        return json.loads(response)

async def generate_with_schema_and_extract(
    model: DeepEvalBaseLLM,
    prompt: str,
    schema: Type[T],
    temperature: float = 0
) -> T:
    """生成并提取结构化输出"""
    response = await model.a_generate(prompt, temperature=temperature)
    json_data = trim_and_load_json(response)
    return schema.model_validate(json_data)
```

from contextlib import contextmanager
from rich.progress import Progress, SpinnerColumn, BarColumn, TextColumn

@contextmanager
def metric_progress_indicator(
    metric: BaseMetric,
    async_mode: bool = True,
    _show_indicator: bool = True,
):
    """评测进度指示器"""
    if not _show_indicator:
        yield
        return

    description = format_metric_description(metric, async_mode)

    with Progress(
        SpinnerColumn(style="rgb(106,0,255)"),
        BarColumn(bar_width=60, complete_style="rgb(106,0,255)"),
        TextColumn("[progress.description]{task.description}"),
        transient=True,  # 完成后清除
    ) as progress:
        progress.add_task(description=description, total=None)
        yield

def format_metric_description(metric: BaseMetric, async_mode: bool) -> str:
    """格式化指标描述"""
    mode = "async" if async_mode else "sync"
    model = metric.evaluation_model or "default"
    return f"[{mode}] {metric.__class__.__name__} ({model})"
```

### 10.1 G-Eval - 通用评测框架

**文件**: `deepeval/metrics/g_eval/g_eval.py`

基于论文 [G-Eval](https://arxiv.org/pdf/2303.16634.pdf) 实现:

```python
class GEval(BaseMetric):
    """通用 LLM 评测 - 自定义评分标准"""

    def __init__(
        self,
        name: str,
        criteria: str,                          # 评测标准描述
        evaluation_steps: List[str],            # 评测步骤
        rubric: Optional[Rubric] = None,        # 评分细则
        evaluation_params: List[LLMTestCaseParams] = None,
    ):
        self.name = name
        self.criteria = criteria
        self.evaluation_steps = evaluation_steps
        self.rubric = rubric or Rubric(score_range=(0, 10))

    async def a_measure(self, test_case: LLMTestCase) -> float:
        # 使用 top_logprobs 获取置信度加权分数
        prompt = self._build_evaluation_prompt(test_case)
        response = await self.model.a_generate(
            prompt,
            logprobs=True,
            top_logprobs=5
        )

        # 加权求和计算最终分数
        score = calculate_weighted_summed_score(response.logprobs)
        self.score = normalize_score(score, self.rubric.score_range)
        return self.score
```

### 10.2 DAG 评测 - 规则图

**文件**: `deepeval/metrics/dag/`

无需 LLM 的确定性评测:

```python
class DAGMetric(BaseMetric):
    """基于有向无环图的规则评测"""

    def __init__(self, dag: EvaluationDAG):
        self.dag = dag

    def measure(self, test_case: LLMTestCase) -> float:
        return self.dag.evaluate(test_case)

# DAG 定义示例
dag = (
    EvaluationDAG()
    .add_condition("has_greeting", lambda tc: "hello" in tc.actual_output.lower())
    .add_condition("has_name", lambda tc: tc.input.split()[0] in tc.actual_output)
    .add_branch("has_greeting", true_score=0.5, false_score=0)
    .add_branch("has_name", true_score=0.5, false_score=0)
)
```

class ArenaTestCase(BaseModel):

### 10.4 成本追踪

```python
class BaseMetric:
    evaluation_cost: Optional[float] = None

    def _accrue_cost(self, cost: float):
        if self.evaluation_cost is None:
            self.evaluation_cost = 0
        self.evaluation_cost += cost

# 在 Provider 中调用
class GPTModel(DeepEvalBaseLLM):
    async def a_generate(self, prompt: str, **kwargs) -> str:
        response = await self.client.chat.completions.create(...)
        cost = calculate_cost(response.usage, self.model_name)
        self._current_metric._accrue_cost(cost)
        return response.choices[0].message.content
```
