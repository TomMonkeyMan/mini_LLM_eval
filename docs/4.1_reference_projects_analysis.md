```python
class AnthropicAdapter(BaseAdapter):
    def instrument(self, session: RecordingSession):
        """Monkey-patch Anthropic client"""
        original_create = anthropic.messages.create

        def patched_create(*args, **kwargs):
            start = time.time()
            response = original_create(*args, **kwargs)
            latency = (time.time() - start) * 1000

            session.add_llm_call(
                model=kwargs.get("model"),
                input_messages=kwargs.get("messages"),
                output_message=response.content,
                latency_ms=latency,
                input_tokens=response.usage.input_tokens,
                output_tokens=response.usage.output_tokens
            )
            return response

        anthropic.messages.create = patched_create

    def uninstrument(self):
        """恢复原始方法"""
        anthropic.messages.create = self._original_create
```

#### 2.2.2 Solver + TaskState 模式

**文件**: `evals/solvers/solver.py`

```python
@dataclass
class TaskState:
    """评测任务的状态"""
    task_description: str  # 固定的任务描述
    messages: list[Message] = field(default_factory=list)  # 对话历史
    current_state: Any = None  # 当前状态快照

class SolverResult:
    output: str
    metadata: dict

class Solver(ABC, CompletionFn):
    """问题求解器抽象"""
    def __init__(self, postprocessors: list[str] = []):
        # 动态加载后处理器
        self.postprocessors = [
            import_class(p)() for p in postprocessors
        ]

    @abstractmethod
    def _solve(self, task_state: TaskState, **kwargs) -> SolverResult:
        pass

    def __call__(self, task_state: TaskState, **kwargs) -> SolverResult:
        result = self._solve(deepcopy(task_state), **kwargs)
        for pp in self.postprocessors:
            result = pp(result)
            record_event("postprocessor", {"name": pp.__class__.__name__})
        return result
```

**借鉴点**:
- 分离问题定义 (TaskState) 和求解方法 (Solver)
- 后处理器链式处理输出
- 支持嵌套 Solver: CoT、Few-shot、Self-consistency

```python
        self._events = []
        self._lock = threading.Lock()
        self._last_flush = time.time()
```

#### 2.2.5 Eval 基类

**文件**: `evals/eval.py`

```python
class Eval(ABC):
    def __init__(self, completion_fns: list[CompletionFn], **kwargs):
        self.completion_fns = completion_fns

    @abc.abstractmethod
    def eval_sample(self, sample: dict, rng: random.Random):
        """评测单个样本，通过 recorder 记录结果"""
        pass

    def run(self, recorder: RecorderBase):
        samples = list(self.get_samples())
        random.Random(SHUFFLE_SEED).shuffle(samples)

        if self.max_samples:
            samples = samples[:self.max_samples]

        for sample in samples:
            with recorder.as_sample(sample["id"]):
                try:
                    self.eval_sample(sample, rng)
                except Exception as e:
                    recorder.record_event("error", {"message": str(e)})
```

**多选题评测示例** (`evals/elsuite/multiple_choice.py`):

#### 2.2.6 指标计算

**文件**: `evals/metrics.py`

```python
def get_accuracy(events: list[Event]) -> float:
    matches = [e for e in events if e.type == "match"]
    return sum(int(e.data["correct"]) for e in matches) / len(matches)

def compute_confusion_matrix(events: list[Event], labels: list[str]) -> np.ndarray:
    matrix = np.zeros((len(labels), len(labels)), dtype=int)
    for event in events:
        if event.type == "match":
            i = labels.index(event.data["expected"])
            j = labels.index(event.data["picked"])
            matrix[i, j] += 1
    return matrix

def compute_precision(cm: np.ndarray, class_idx: int = 0) -> float:
    return cm[class_idx, class_idx] / cm[:, class_idx].sum()

def compute_recall(cm: np.ndarray, class_idx: int = 0) -> float:
    return cm[class_idx, class_idx] / cm[class_idx, :].sum()

def compute_f1(cm: np.ndarray, class_idx: int = 0, beta: float = 1.0) -> float:
    p = compute_precision(cm, class_idx)
    r = compute_recall(cm, class_idx)
    return (1 + beta**2) * p * r / (beta**2 * p + r)
```

**借鉴点**:
- 从事件流计算聚合指标
- 支持混淆矩阵分析
- 可扩展指标函数

```python
def get_jsonl(filename: str) -> Iterator[dict]:
    """流式读取 JSONL，支持多种格式"""
    with open_by_file_pattern(filename) as f:
        for line in f:
            yield json.loads(line)

def open_by_file_pattern(filename: str, mode: str = "r"):
    """支持多种压缩格式和远程路径"""
    if filename.endswith(".gz"):
        return gzip.open(filename, mode + "t")
    elif filename.endswith(".lz4"):
        return lz4.frame.open(filename, mode)
    elif filename.endswith(".zst"):
        return zstandard.open(filename, mode)
    elif filename.startswith("gs://"):
        return blobfile.BlobFile(filename, mode)
    elif filename.startswith("hf://"):
        return load_huggingface_dataset(filename)
    else:
        return open(filename, mode)
```

#### 3.2.2 LLM-as-Judge 实现

**文件**: `ragas/metrics/_faithfulness.py`

```python
class Faithfulness(MetricWithLLM, SingleTurnMetric):
    """两步法忠实度评测"""

    async def _ascore(self, row: SingleTurnSample) -> float:
        # Step 1: 从回答中提取陈述
        statements = await self._extract_statements(row.response)

        # Step 2: 对每个陈述进行 NLI 验证
        verdicts = []
        for stmt in statements:
            verdict = await self._verify_statement(
                statement=stmt,
                context=row.retrieved_contexts
            )
            verdicts.append(verdict)

        # 计算忠实度得分
        faithful_count = sum(1 for v in verdicts if v.verdict == "yes")
        return faithful_count / len(verdicts) if verdicts else 0.0
```

**借鉴点**:
- 分步评测 (提取 → 验证)
- 细粒度判断 (每个陈述独立评分)
- 结构化 LLM 输出

```python
class PydanticPrompt(BasePrompt, Generic[InputModel, OutputModel]):
    """强类型 Prompt"""
    input_model: Type[InputModel]
    output_model: Type[OutputModel]
    instruction: str
    examples: List[Tuple[InputModel, OutputModel]] = []

    async def generate(
        self,
        llm: BaseRagasLLM,
        data: InputModel,
        temperature: Optional[float] = None,
        retries_left: int = 3,
    ) -> OutputModel:
        # 使用 instructor 库强制结构化输出
        prompt = self._build_prompt(data)
        response = await llm.agenerate_text(prompt, temperature=temperature)
        return self.output_model.model_validate_json(response)
```

**Few-Shot 动态选择**:
```python
class DynamicFewShotPrompt(PydanticPrompt):
    """基于相似度的动态 Few-Shot"""
    example_store: List[Tuple[InputModel, OutputModel]]
    embeddings: BaseRagasEmbeddings

    def _select_examples(self, input_data: InputModel, k: int = 3) -> List:
        # 计算输入与示例的相似度
        input_embedding = self.embeddings.embed(str(input_data))
        similarities = [
            (ex, cosine_similarity(input_embedding, self.embeddings.embed(str(ex[0]))))
            for ex in self.example_store
        ]
        # 返回最相似的 k 个示例
        return [ex for ex, _ in sorted(similarities, key=lambda x: -x[1])[:k]]
```

**借鉴点**:
- Pydantic 定义输入输出 Schema
- 自动 JSON Schema 生成

#### 3.2.5 异步执行器

**文件**: `ragas/executor.py`

```python
@dataclass
class Executor:
    desc: str = "Evaluating"
    show_progress: bool = True
    batch_size: Optional[int] = None
    raise_exceptions: bool = False
    jobs: List[Any] = field(default_factory=list)

    def submit(
        self,
        callable: Callable,
        *args,
        name: Optional[str] = None,
        **kwargs,
    ) -> int:
        """提交任务，返回索引"""
        job_id = len(self.jobs)
        self.jobs.append((callable, args, kwargs, name, job_id))
        return job_id

    async def aresults(self) -> List[Any]:
        """异步执行所有任务"""
        results = [None] * len(self.jobs)

        async def run_job(job):
            callable, args, kwargs, name, job_id = job
            try:
                result = await callable(*args, **kwargs)
                results[job_id] = result
            except Exception as e:
                if self.raise_exceptions:
                    raise
                results[job_id] = float("nan")

        with tqdm(total=len(self.jobs), desc=self.desc) as pbar:
            for batch in self._batched(self.jobs, self.batch_size):
                await asyncio.gather(*[run_job(j) for j in batch])
                pbar.update(len(batch))

        return results

    def results(self) -> List[Any]:
        """同步包装"""
        return asyncio.run(self.aresults())
```

**借鉴点**:
- 任务提交与执行分离
- 批量处理 + 进度条
- 异常隔离 (返回 NaN)

#### 3.2.7 缓存机制

**文件**: `ragas/cache.py`

```python
class CacheInterface(ABC):
    @abstractmethod
    def get(self, key: str) -> Any: ...
    @abstractmethod
    def set(self, key: str, value: Any) -> None: ...
    @abstractmethod
    def has_key(self, key: str) -> bool: ...

class DiskCacheBackend(CacheInterface):
    """持久化磁盘缓存"""
    def __init__(self, cache_dir: str = ".ragas_cache"):
        self._cache = diskcache.Cache(cache_dir)

def cacher(cache_backend: Optional[CacheInterface] = None):
    """缓存装饰器"""
    def decorator(func):
        @functools.wraps(func)
        async def async_wrapper(*args, **kwargs):
            key = _generate_key(func.__name__, args, kwargs)
            if cache_backend and cache_backend.has_key(key):
                return cache_backend.get(key)
            result = await func(*args, **kwargs)
            if cache_backend:
                cache_backend.set(key, result)
            return result
        return async_wrapper
    return decorator
```

**借鉴点**:
- 装饰器缓存 LLM 调用
- SHA256 键生成
- 可插拔缓存后端

#### 3.2.10 评测入口 API

**文件**: `ragas/evaluation.py`

```python
async def aevaluate(
    dataset: Union[Dataset, EvaluationDataset],
    metrics: Optional[Sequence[Metric]] = None,
    llm: Optional[BaseRagasLLM] = None,
    embeddings: Optional[BaseRagasEmbeddings] = None,
    experiment_name: Optional[str] = None,
    run_config: Optional[RunConfig] = None,
    show_progress: bool = True,
    batch_size: Optional[int] = None,
    raise_exceptions: bool = False,
) -> EvaluationResult:
    """异步评测入口"""

    # 初始化默认指标
    if metrics is None:
        metrics = [Faithfulness(), AnswerRelevancy(), ContextPrecision()]

    # 注入共享 LLM/Embeddings
    for metric in metrics:
        if isinstance(metric, MetricWithLLM) and metric.llm is None:
            metric.llm = llm or llm_factory()
        if isinstance(metric, MetricWithEmbeddings) and metric.embeddings is None:
            metric.embeddings = embeddings or embedding_factory()

    # 执行评测
    executor = Executor(desc="Evaluating", show_progress=show_progress, batch_size=batch_size)

    for sample in dataset.samples:
        for metric in metrics:
            executor.submit(metric._ascore, sample, name=f"{metric.name}")

    scores = await executor.aresults()

    return EvaluationResult(scores=scores, dataset=dataset, metrics=metrics)

def evaluate(...) -> EvaluationResult:
    """同步入口"""
    return asyncio.run(aevaluate(...))
```

**借鉴点**:
- 异步优先，同步包装
- 默认指标集
- 共享依赖注入

## 5. 针对 Mini LLM Eval 的借鉴建议

### 5.1 必须采用

1. **Pydantic 数据模型** - 强类型、自动校验、序列化 (三个项目都用)
2. **Protocol 定义 Provider 接口** - 松耦合、易扩展 (Evals)
3. **YAML 配置 + 动态加载** - 声明式、可热更新 (AgentProbe + Evals)
4. **JSONL 事件流记录** - 流式写入、便于分析 (Evals)
5. **装饰器注册 Evaluator** - 简洁、自动发现 (AgentProbe)
6. **原生 async/await** - 异步优先，同步包装 (RAGAS)
7. **Executor 任务执行器** - 批量并发、进度追踪 (RAGAS)

### 5.2 推荐采用

1. **LLM 调用缓存** - 装饰器 + 磁盘缓存 (RAGAS)
2. **成本计算器** - 独立定价表，支持配置覆盖 (AgentProbe)
3. **Rich CLI 输出** - 进度条、表格、美观 (AgentProbe)
4. **Mixin 组合指标** - MetricWithLLM、MetricWithEmbeddings (RAGAS)
5. **Pydantic Prompt** - 结构化 LLM 输出 (RAGAS)
6. **RunConfig 运行配置** - 重试、超时、并发数 (RAGAS)
7. **混淆矩阵指标** - Precision/Recall/F1 (Evals)

### 5.3 可选采用

1. **Adapter 拦截模式** - 如需无侵入追踪 (AgentProbe)
2. **Solver + TaskState** - 如需支持多轮对话评测 (Evals)
3. **多压缩格式** - 如数据集较大 (Evals)
4. **动态 Few-Shot** - 基于相似度选择示例 (RAGAS)
5. **存储后端插件** - Entry Points 机制 (RAGAS)

#### LLM 调用缓存 (来自 RAGAS)
```python
# 参考 ragas/cache.py
import hashlib
import diskcache

class DiskCacheBackend:
    def __init__(self, cache_dir: str = ".eval_cache"):
        self._cache = diskcache.Cache(cache_dir)

    def get(self, key: str) -> Any:
        return self._cache.get(key)

    def set(self, key: str, value: Any) -> None:
        self._cache.set(key, value)

    def has_key(self, key: str) -> bool:
        return key in self._cache

def cacher(cache_backend: Optional[DiskCacheBackend] = None):
    def decorator(func):
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            if cache_backend is None:
                return await func(*args, **kwargs)

            key = hashlib.sha256(
                f"{func.__name__}:{args}:{kwargs}".encode()
            ).hexdigest()

            if cache_backend.has_key(key):
                return cache_backend.get(key)

            result = await func(*args, **kwargs)
            cache_backend.set(key, result)
            return result
        return wrapper
    return decorator
```

#### Mixin 组合指标 (来自 RAGAS)
```python
# 参考 ragas/metrics/base.py
class Metric(ABC):
    name: str
    _required_columns: dict[str, set[str]]

    @abstractmethod
    async def _ascore(self, sample: dict) -> float:
        pass

class MetricWithLLM(Metric):
    llm: Optional[BaseRagasLLM] = None

class MetricWithEmbeddings(Metric):
    embeddings: Optional[BaseRagasEmbeddings] = None

# 组合使用
class Faithfulness(MetricWithLLM):
    """忠实度评测 - 需要 LLM"""
    name = "faithfulness"
    _required_columns = {"response", "retrieved_contexts"}

    async def _ascore(self, sample: SingleTurnSample) -> float:
        # 两步评测：提取陈述 → NLI 验证
        statements = await self._extract_statements(sample.response)
        verdicts = [await self._verify(s, sample.retrieved_contexts) for s in statements]
        return sum(v == "yes" for v in verdicts) / len(verdicts)
```
